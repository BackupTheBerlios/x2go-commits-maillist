+if [ "$cmd" == "GNOME" ]; then

+	# Ubuntu starting with 12.04 (precise) launches Unity when calling gnome-session.

+	if [ "$DISTRIB_ID" == "Ubuntu" ] && echo "$DISTRIB_CODENAME" | egrep -v "^[a-o].*" >/dev/null; then
